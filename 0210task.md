import CompAction from "comps/actions";
import { Comp } from "comps/baseComps";
import { OptionalComp } from "comps/baseComps/comp";
import { evalAndReduce } from "comps/utils";
import { Tabs, Bottom, BottomLeft, BottomRight } from "design";
import Header, { Panelstatus } from "pages/common/header";
import React from "react";
import { GlobalStateManager } from "state";
import { GridLayoutComp, COMP_TYPES } from "./comps/comps/gridLayoutComp";
import { CompContext } from "./comps/utils";
import {
  Body,
  Height100Div,
  LeftPanel,
  MiddlePanel,
  EditorContainer,
  BottomContainer,
  RightPanel,
  Resize,
} from "./pages";
import LeftPanelContainer from "./pages/leftPanel/leftPanelContainer";

const VariantComp = GridLayoutComp;
class App extends React.Component {
  state: {
    comp: Comp;
    showPropertyPane: boolean;
    selectedPath?: Array<string>;
    isDragging: boolean;
    showPanels: Panelstatus;
  };
  constructor(props: any) {
    super(props);
    this.dispatch = this.dispatch.bind(this);
    const compStr = localStorage.getItem("main_comp");
    let comp = new VariantComp({ dispatch: this.dispatch });
    if (compStr) {
      const defaultValue = JSON.parse(compStr);
      comp = new VariantComp({ dispatch: this.dispatch, value: defaultValue });
    }
    // console.log("after " + JSON.stringify(comp));
    console.log(comp.toJsonValue());
    comp = evalAndReduce(comp);
    GlobalStateManager.DEFAULT.setComp(comp);
    this.state = {
      comp: comp,
      showPropertyPane: false,
      isDragging: false,
      showPanels: { left: true, middle: true, right: true },
    };
    this.showLeft = this.showLeft.bind(this);
    this.showMiddle = this.showMiddle.bind(this);
    this.showRight = this.showRight.bind(this);
    this.resizeBottom = this.resizeBottom.bind(this);
  }
  drag = React.createRef<HTMLDivElement>();
  bottom = React.createRef<HTMLDivElement>();
  resizeBottom(e: any) {
    var startY = e.clientY;
    var _this = this;
    const curH = (_this.bottom.current as HTMLDivElement).clientHeight;
    e.stopPropagation();
    e.preventDefault();
    document.onmousemove = function (e) {
      e.stopPropagation();
      e.preventDefault();
      var endY = e.clientY;
      var moveLen = startY - endY;
      (_this.bottom.current as HTMLDivElement).style.height =
        curH + moveLen + "px";
    };
    document.onmouseup = () => {
      document.onmousemove = null;
      document.onmouseup = null;
    };
  }
  dispatch(action: CompAction) {
    if (action.type === "SELECT_COMP") {
      // console.log("select " + JSON.stringify(action));
      this.setState((state) => {
        const comp = this.state.comp;
        const selectedComp = comp.selectChild(action.path);
        const hasComp = selectedComp && selectedComp !== comp;
        return { selectedPath: action.path, showPropertyPane: hasComp };
      });
      return;
    }
    console.log("receive action " + JSON.stringify(action));
    this.setState((state: { comp: Comp }) => {
      let comp = state.comp.reduce(action);
      // console.log(comp.getView());
      // console.log("after action " + JSON.stringify(action) + ", " + JSON.stringify(comp));
      comp = evalAndReduce(comp);
      // 这儿是插入中间件的地方，可能的功能有:
      // Evaluation、Persistence、UndoRedo、MultiUser
      GlobalStateManager.DEFAULT.setComp(comp);
      localStorage.setItem("main_comp", JSON.stringify(comp.toJsonValue()));
      return { comp: comp };
    });
  }
  showLeft() {
    this.setState((state: { showPanels: Panelstatus }) => {
      let status = { ...state.showPanels, left: !state.showPanels.left };
      return { showPanels: status };
    });
  }
  showMiddle() {
    this.setState((state: { showPanels: Panelstatus }) => {
      let status = { ...state.showPanels, middle: !state.showPanels.middle };
      return { showPanels: status };
    });
  }
  showRight() {
    this.setState((state: { showPanels: Panelstatus }) => {
      let status = { ...state.showPanels, right: !state.showPanels.right };
      return { showPanels: status };
    });
  }
  render() {
    const comp = this.state.comp;
    let selectedComp: OptionalComp;
    if (this.state.selectedPath) {
      selectedComp = comp.selectChild(this.state.selectedPath);
      if (selectedComp === comp) {
        // hack，之后用更好的方式表达反选
        selectedComp = undefined;
      }
    }
    const setDragging = (dragging: boolean) => {
      this.setState({ isDragging: dragging });
    };
    const showPropertyPane = this.state.showPropertyPane;
    function rightPanelView() {
      if (showPropertyPane) {
        if (selectedComp) {
          return selectedComp.getPropertyView();
        } else {
          return <div>没有组件被选择，点击选择组件</div>;
        }
      }
      return COMP_TYPES.map((compType) => (
        <div
          draggable
          onDragStart={(e) => {
            e.dataTransfer.setData("compType", compType);
            setDragging(true);
          }}
        >
          {compType}
        </div>
      ));
    }
    return (
      <Height100Div
        onDragEnd={(e) => {
          setDragging(false);
        }}
      >
        <Header
          setShow={[this.showLeft, this.showMiddle, this.showRight]}
          curStatus={this.state.showPanels}
        />
        <Body>
          <LeftPanel show={this.state.showPanels.left}>
            <LeftPanelContainer />
          </LeftPanel>
          <MiddlePanel>
            <EditorContainer>
              <CompContext.Provider
                value={{
                  selectedComp: selectedComp,
                  isDragging: this.state.isDragging,
                  setDragging: setDragging,
                }}
              >
                {comp.getView()}
              </CompContext.Provider>
            </EditorContainer>
            <Resize onMouseDown={this.resizeBottom} ref={this.drag} />
            <BottomContainer
              show={this.state.showPanels.middle}
              ref={this.bottom}
            >
              <Bottom>
                <BottomLeft />
                <BottomRight />
              </Bottom>
            </BottomContainer>
          </MiddlePanel>
          <RightPanel show={this.state.showPanels.right}>
            <Tabs
              onChange={(key) =>
                this.setState({ showPropertyPane: key === "property" })
              }
              tabsConfig={[
                { key: "property", title: "属性" },
                { key: "insert", title: "插入" },
              ]}
              activeKey={this.state.showPropertyPane ? "property" : "insert"}
            >
              {rightPanelView()}
            </Tabs>
          </RightPanel>
        </Body>
      </Height100Div>
    );
  }
}

export default App;
