//创建部门表
create table bumen_sheet(
bumen_num int primary key,
bumen_name varchar(20) unique not null,
bumen_renshu int(10) not null
);

//创建员工表
create table yuangong_sheet (
	number int not null primary key,
	name varchar(20) not null,
	age int CHECK(age>=1) not null,
	sex varchar(2) CHECK(sex in('男','女')) not null,
	birthday Datetime not null,
	martial varchar(4) CHECK(martial in ('已婚','未婚')),
	political varchar(20) CHECK(political in ('群众','团员','党员')) not null,
	schoollevel varchar(20)not null,
	bumen_num int nut null ,
	position varchar(20) not null,
	contact varchar(11) unique not null,
	note varchar(100),
	foreign key (bumen_num) references bumen_sheet(bumen_num)
)
//创建薪资表
create table salary_sheet(
salary_num int primary key,
number int not null,
basic_salary int CHECK(basic_salary>0) not null,
welfare int CHECK(welfare>0) not null,
bonus int CHECK(bonus>0) not null,
calculation_method varchar(50) not null,
real_wages int CHECK(real_wages>0),
issuing_date Datetime not null,
foreign key (number) references yuangong_sheet(number)
)
//创建奖惩表
create table rd_sheet(
	rd_num int primary key,
  number int not null,
  rd_time Datetime not null,
  rd_why varchar(100) not null,
  rd_position varchar(20) not null,
  note varchar(500),
  foreign key (number) references yuangong_sheet(number)
)

//视图员工信息
create view basic_information(yuangong_name,bumen_name,position,contact,real_wages)
as select name,bumen_name,position,contact,real_wages 
from yuangong_sheet,bumen_sheet,salary_sheet
where yuangong_sheet.number=salary_sheet.number and yuangong_sheet.bumen_num=bumen_sheet.bumen_num

//创建存储过程统计各部门各种职称的人数
create procedure statistic
@ID INT out
as select bumen_sheet.bumen_num,bumen_sheet.bumen_name,yuangong_sheet.number,yuangong_sheet.name,@ID=count(*)
from yuangong_sheet,bumen_sheet
where yuangong_sheet.bumen_num=bumen_sheet.bumen_num and yuangong_sheet.position='教授' 


//存储过程
delimiter
create procedure statistic(in numb varchar(10))
begin
	select count(*) from yuangong_sheet where yuangong_sheet.position = numb group by yuangong_sheet.position ;
end
delimiter;
//调用 
call statistic('要查找的内容')

//创建触发器，实现修改教师所在部门时自动修改相应部门的教师人数
delimiter//
create trigger count_staff after insert on yuangong_sheet 
for each row
begin
update bumen_sheet 
set bumen_sheet.bumen_renshu=bumen_sheet.bumen_renshu+1
where bumen_sheet.bumen_num=(select bumen_sheet.bumen_num from Inserted)
end//
delimiter;
